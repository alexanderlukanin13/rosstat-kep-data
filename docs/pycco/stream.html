def yield_flat_tuples(row_tuple):

    

  

  

  

    

      

        #
      

      
    

    

      
    vn, y, a, qs, ms = row_tuple
        
    if a is not None:
               yield ("a", vn, y, SAFE_NONE, SAFE_NONE, a)

    if qs is not None:         
        for i, val in enumerate(qs):
            if val is not None:
               yield ("q", vn, y, i+1, SAFE_NONE, val)

    if ms is not None:         
        for j, val in enumerate(ms):
            if val is not None:
               yield ("m", vn, y, SAFE_NONE, j+1, val)
